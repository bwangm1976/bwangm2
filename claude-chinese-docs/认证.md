## 认证方式

### 1. API 密钥认证

Claude API 使用 API 密钥进行认证。这是最常用的认证方式。

#### 获取 API 密钥

1. 登录 [Anthropic 控制台](https://console.anthropic.com)
2. 导航到 "API Keys" 部分
3. 点击 "Create new API key"
4. 输入密钥名称和描述
5. 复制生成的 API 密钥

**注意：** API 密钥只显示一次，请务必安全保存。

#### 使用 API 密钥

```python
import anthropic

# 方法1：直接传入
```

```python
# secrets_manager.py
import os
import json
from pathlib import Path

class SecretsManager:
    """密钥管理器"""

    def __init__(self, key_file: str = None):
        self.key_file = key_file or Path.home() / '.claude' / 'secrets.json'
        self.secrets = self._load_secrets()

    def _load_secrets(self):
        """加载密钥"""
        if self.key_file.exists():
            with open(self.key_file, 'r') as f:
                return json.load(f)
        return {}

    def _save_secrets(self):
        """保存密钥"""
        self.key_file.parent.mkdir(parents=True, exist_ok=True)
        with open(self.key_file, 'w') as f:
            json.dump(self.secrets, f, indent=2)

    def set_secret(self, key: str, value: str):
        """设置密钥"""
        self.secrets[key] = value
        self._save_secrets()

    def get_secret(self, key: str) -> str:
        """获取密钥"""
        return self.secrets.get(key)

    def delete_secret(self, key: str):
        """删除密钥"""
        if key in self.secrets:
            del self.secrets[key]
            self._save_secrets()

# 使用示例
secrets = SecretsManager()
secrets.set_secret('anthropic_api', 'your-api-key')
api_key = secrets.get_secret('anthropic_api')
```

### 2. 加密存储

```python
from cryptography.fernet import Fernet
import os
from pathlib import Path

class EncryptedSecretsManager:
    """加密密钥管理器"""

    def __init__(self, key_file: str = None, password: str = None):
        self.key_file = key_file or Path.home() / '.claude' / '.key'
        self.password = password or os.getenv('CLAUDE_PASSWORD')

        # 生成或加载加密密钥
        if self.key_file.exists():
            self._load_key()
        else:
            self._generate_key()

    def _generate_key(self):
        """生成加密密钥"""
        key = Fernet.generate_key()
        with open(self.key_file, 'wb') as f:
            f.write(key)

        # 如果提供了密码，使用密码加密密钥文件
        if self.password:
            self._encrypt_key_file()

    def _load_key(self):
        """加载加密密钥"""
        with open(self.key_file, 'rb') as f:
            key = f.read()

        # 如果有密码，先解密
        if self.password:
            key = self._decrypt_data(key)

        self.cipher = Fernet(key)

    def _encrypt_key_file(self):
        """加密密钥文件"""
        with open(self.key_file, 'rb') as f:
            data = f.read()

        encrypted = self._encrypt_data(data)
        with open(self.key_file, 'wb') as f:
            f.write(encrypted)

    def _decrypt_key_file(self):
        """解密密钥文件"""
        with open(self.key_file, 'rb') as f:
            encrypted = f.read()

        return self._decrypt_data(encrypted)

    def _encrypt_data(self, data):
        """加密数据"""
        return self.cipher.encrypt(data)

    def _decrypt_data(self, encrypted_data):
        """解密数据"""
        # 这里应该使用密码来解密实际的 Fernet 密钥
        # 简化示例
        return Fernet.generate_key()  # 实际应该使用密码解密

    def get_secret(self, key: str) -> str:
        """获取加密的密钥"""
        encrypted_value = os.getenv(key)
        if encrypted_value:
            return self.cipher.decrypt(encrypted_value.encode()).decode()
        return None

# 使用示例

    password="your-password"
)
```

## 身份验证流程

### 1. 完整认证流程

```python
import anthropic
import time
from anthropic import APIError, AuthenticationError, RateLimitError

class AuthenticatedClient:
    """带认证的客户端"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.client = anthropic.Anthropic(api_key=api_key)
        self.last_request_time = 0
        self.rate_limit_delay = 1  # 秒

    def _check_rate_limit(self):
        """检查速率限制"""
        current_time = time.time()
        if current_time - self.last_request_time < self.rate_limit_delay:
            time.sleep(self.rate_limit_delay - (current_time - self.last_request_time))
        self.last_request_time = time.time()

    def authenticate(self) -> bool:
        """验证 API 密钥"""
        try:
            self._check_rate_limit()

            # 发送测试请求
            response = self.client.messages.create(
                model="claude-3-haiku-20240229",
                max_tokens=1,
                messages=[{"role": "user", "content": "test"}]
            )
            return True

        except AuthenticationError:
            print("API 密钥无效")
            return False
        except APIError as e:
            print(f"认证错误: {e}")
            return False

    def send_request(self, messages, **kwargs):
        """发送带认证的请求"""
        if not self.authenticate():
            raise AuthenticationError("Authentication failed")

        self._check_rate_limit()

        try:
            response = self.client.messages.create(
                messages=messages,
                **kwargs
            )
            return response

        except RateLimitError as e:
            print(f"速率限制错误: {e}")
            time.sleep(5)  # 等待后重试
            return self.send_request(messages, **kwargs)
        except APIError as e:
            print(f"API 错误: {e}")
            raise
```

### 2. 多重认证

```python
class MultiAuthClient:
    """多重认证客户端"""

    def __init__(self):
        self.keys = []
        self.current_key_index = 0

    def add_key(self, api_key: str):
        """添加 API 密钥"""
        self.keys.append(api_key)

    def get_next_key(self):
        """获取下一个可用的 API 密钥"""
        if not self.keys:
            raise ValueError("No API keys available")

        key = self.keys[self.current_key_index]
        self.current_key_index = (self.current_key_index + 1) % len(self.keys)
        return key

    def authenticate_and_request(self, messages, **kwargs):
        """使用可用的密钥进行认证和请求"""
        max_attempts = len(self.keys)

        for attempt in range(max_attempts):
            try:
                api_key = self.get_next_key()
                client = anthropic.Anthropic(api_key=api_key)

                response = client.messages.create(
                    messages=messages,
                    **kwargs
                )

                return response

            except AuthenticationError:
                print(f"API 密钥无效，尝试下一个...")
                continue
            except RateLimitError:
                print(f"达到速率限制，等待...")
                time.sleep(2 ** attempt)  # 指数退避
                continue

        raise AuthenticationError("All API keys failed")
```

```python
import logging
from datetime import datetime
from typing import Dict, Any

class AuditLogger:
    """审计日志记录器"""

    def __init__(self, log_file: str = 'audit.log'):
        self.logger = logging.getLogger('audit')
        self.logger.setLevel(logging.INFO)

        # 创建文件处理器
        handler = logging.FileHandler(log_file)
        formatter = logging.Formatter(
            '%(asctime)s - %(levelname)s - %(message)s'
        )
        handler.setFormatter(formatter)
        self.logger.addHandler(handler)

    def log_request(self, api_key: str, endpoint: str, params: Dict[str, Any]):
        """记录 API 请求"""
        self.logger.info(
            f"API Request - Key: {self._mask_key(api_key)}, "
            f"Endpoint: {endpoint}, "
            f"Params: {self._mask_params(params)}"
        )

    def log_response(self, api_key: str, status: str, duration: float):
        """记录 API 响应"""
        self.logger.info(
            f"API Response - Key: {self._mask_key(api_key)}, "
            f"Status: {status}, "
            f"Duration: {duration:.2f}s"
        )

    def log_error(self, api_key: str, error: Exception):
        """记录错误"""
        self.logger.error(
            f"API Error - Key: {self._mask_key(api_key)}, "
            f"Error: {str(error)}"
        )

    def _mask_key(self, key: str) -> str:
        """掩码 API 密钥"""
        if len(key) > 8:
            return f"{key[:4]}...{key[-4:]}"
        return "***"

    def _mask_params(self, params: Dict[str, Any]) -> Dict[str, Any]:
        """掩码敏感参数"""
        masked = params.copy()
        if 'messages' in masked:
            # 替换消息内容为长度
            if isinstance(masked['messages'], list):
                masked['messages'] = [
                    {"role": msg.get('role'), "content": f"[{len(str(msg.get('content', '')))} chars]"}
                    for msg in masked['messages']
                ]
        return masked

# 使用示例
```

## 错误处理

### 1. 认证错误处理

```python
from anthropic import AuthenticationError, RateLimitError, APIError
import time

class AuthErrorHandler:
    """认证错误处理器"""

    def __init__(self, max_retries: int = 3):
        self.max_retries = max_retries

    def handle_auth_error(self, error: Exception, api_key: str) -> str:
        """处理认证错误"""
        if isinstance(error, AuthenticationError):
            return "无效的 API 密钥，请检查密钥是否正确"
        elif isinstance(error, RateLimitError):
            return "请求过于频繁，请稍后再试"
        elif isinstance(error, APIError):
            if "quota" in str(error).lower():
                return "已达到 API 使用限制"
            elif "forbidden" in str(error).lower():
                return "没有权限访问此资源"
            else:
                return f"API 错误: {str(error)}"
        else:
            return f"未知错误: {str(error)}"

    def retry_with_backoff(self, func, *args, **kwargs):
        """指数退避重试"""
        for attempt in range(self.max_retries):
            try:
                return func(*args, **kwargs)
            except (AuthenticationError, RateLimitError, APIError) as e:
                if attempt == self.max_retries - 1:
                    raise

                # 计算等待时间（指数退避）
                wait_time = (2 ** attempt) + (attempt * 0.5)
                print(f"尝试 {attempt + 1}/{self.max_retries} 失败，{wait_time:.1f}秒后重试...")
                time.sleep(wait_time)

# 使用示例
error_handler = AuthErrorHandler()
try:
    response = error_handler.retry_with_backoff(
        client.messages.create,
        messages=[{"role": "user", "content": "Hello"}],
        max_tokens=1000
    )
except Exception as e:
    error_message = error_handler.handle_auth_error(e, api_key)
    print(error_message)
```

### 2. 密钥轮换

```python
import random
from datetime import datetime, timedelta

class KeyRotator:
    """API 密钥轮换器"""

    def __init__(self, keys: list, rotation_period: timedelta = timedelta(days=30)):
        self.keys = keys
        self.rotation_period = rotation_period
        self.current_key_index = 0
        self.last_rotation = datetime.now()

    def get_current_key(self) -> str:
        """获取当前密钥"""
        # 检查是否需要轮换
        if datetime.now() - self.last_rotation > self.rotation_period:
            self.rotate_key()

        return self.keys[self.current_key_index]

    def rotate_key(self):
        """轮换密钥"""
        # 随机选择下一个密钥
        available_indices = [i for i in range(len(self.keys)) if i != self.current_key_index]
        self.current_key_index = random.choice(available_indices)
        self.last_rotation = datetime.now()

        print(f"已轮换到密钥索引 {self.current_key_index}")

    def add_key(self, key: str):
        """添加新密钥"""
        if key not in self.keys:
            self.keys.append(key)

    def remove_key(self, key: str):
        """移除密钥"""
        if key in self.keys:
            self.keys.remove(key)
            # 调整当前索引
            if self.current_key_index >= len(self.keys):
                self.current_key_index = 0

# 使用示例
keys = ["key1", "key2", "key3"]
key_rotator = KeyRotator(keys, rotation_period=timedelta(days=7))

# 获取当前密钥
current_key = key_rotator.get_current_key()
client = anthropic.Anthropic(api_key=current_key)
```

```python
import prometheus_client
from prometheus_client import Counter, Histogram, Gauge

# 定义监控指标
REQUEST_COUNT = Counter('claude_api_requests_total', 'Total API requests')
ERROR_COUNT = Counter('claude_api_errors_total', 'Total API errors', ['error_type'])
REQUEST_DURATION = Histogram('claude_api_request_duration_seconds', 'API request duration')
ACTIVE_KEYS = Gauge('claude_api_active_keys', 'Number of active API keys')

class MonitoredClient:
    """带监控的客户端"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.client = anthropic.Anthropic(api_key=api_key)

        # 更新活跃密钥数
        ACTIVE_KEYS.inc()

    def __del__(self):
        # 减少活跃密钥数
        ACTIVE_KEYS.dec()

    def send_request(self, messages, **kwargs):
        """发送带监控的请求"""
        REQUEST_COUNT.inc()

        start_time = time.time()

        try:
            response = self.client.messages.create(
                messages=messages,
                **kwargs
            )

            # 记录请求时长
            REQUEST_DURATION.observe(time.time() - start_time)

            return response

        except Exception as e:
            # 记录错误
            ERROR_COUNT.labels(error_type=type(e).__name__).inc()
            raise
```

```python
# alerts.py
import smtplib
from email.mime.text import MIMEText
from datetime import datetime

class AlertManager:
    """告警管理器"""

    def __init__(self, email_config: dict):
        self.email_config = email_config
        self.last_alert_time = {}
        self.alert_cooldown = 300  # 5分钟冷却

    def send_alert(self, message: str, alert_type: str = "error"):
        """发送告警"""
        current_time = datetime.now()

        # 检查冷却时间
        if alert_type in self.last_alert_time:
            if (current_time - self.last_alert_time[alert_type]).seconds < self.alert_cooldown:
                return

        self.last_alert_time[alert_type] = current_time

        # 发送邮件告警
        try:
            msg = MIMEText(message)
            msg['Subject'] = f"Claude API Alert - {alert_type}"
            msg['From'] = self.email_config['from']
            msg['To'] = self.email_config['to']

            with smtplib.SMTP(self.email_config['smtp_server']) as server:
                server.send_message(msg)

        except Exception as e:
            print(f"Failed to send alert: {e}")

    def check_api_health(self, client: anthropic.Anthropic):
        """检查 API 健康状态"""
        try:
            # 发送健康检查请求
            response = client.messages.create(
                model="claude-3-haiku-20240229",
                max_tokens=1,
                messages=[{"role": "user", "content": "ping"}]
            )

            return True

        except Exception as e:
            alert_message = f"API health check failed: {str(e)}"
            self.send_alert(alert_message, "health_check")
            return False
```
