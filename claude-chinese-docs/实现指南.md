#### API 客户端封装 (src/api/client.py)

```python
import os
from typing import List, Dict, Any, Optional
from anthropic import Anthropic, APIError, RateLimitError
import logging
from datetime import datetime

class ClaudeClient:
    """Claude API 客户端封装"""

    def __init__(self, api_key: str = None, model: str = "claude-3-sonnet-20240229"):
        self.api_key = api_key or os.getenv("ANTHROPIC_API_KEY")
        if not self.api_key:
            raise ValueError("API key is required")

        self.client = Anthropic(api_key=self.api_key)
        self.model = model
        self.logger = logging.getLogger(__name__)

        # 配置
        self.max_retries = 3
        self.timeout = 30
        self.rate_limit_delay = 1

    def create_message(
        self,
        content: str,
        system: str = None,
        max_tokens: int = 1000,
        temperature: float = 0.7,
        stream: bool = False,
        tools: List[Dict] = None
    ) -> Dict[str, Any]:
        """创建消息"""
        try:
            messages = [{"role": "user", "content": content}]

            kwargs = {
                "model": self.model,
                "max_tokens": max_tokens,
                "messages": messages,
                "temperature": temperature,
                "stream": stream
            }

            if system:
                kwargs["system"] = system
            if tools:
                kwargs["tools"] = tools

            if stream:
                return self._handle_stream(kwargs)
            else:
                response = self.client.messages.create(**kwargs)
                return self._format_response(response)

        except APIError as e:
            self.logger.error(f"API Error: {e}")
            raise
        except Exception as e:
            self.logger.error(f"Unexpected error: {e}")
            raise

    def _handle_stream(self, kwargs) -> Dict[str, Any]:
        """处理流式响应"""
        stream = self.client.messages.create(**kwargs)
        chunks = []

        for chunk in stream:
            if hasattr(chunk, 'delta') and chunk.delta and hasattr(chunk.delta, 'text'):
                chunks.append(chunk.delta.text)

        return {
            "content": "".join(chunks),
            "stream": True,
            "timestamp": datetime.now().isoformat()
        }

    def _format_response(self, response) -> Dict[str, Any]:
        """格式化响应"""
        return {
            "content": response.content[0].text if response.content else "",
            "model": response.model,
            "usage": {
                "input_tokens": response.usage.input_tokens,
                "output_tokens": response.usage.output_tokens,
                "total_tokens": response.usage.input_tokens + response.usage.output_tokens
            },
            "timestamp": datetime.now().isoformat()
        }

    def retry_with_backoff(self, func, *args, **kwargs):
        """指数退避重试"""
        import time

        for attempt in range(self.max_retries):
            try:
                return func(*args, **kwargs)
            except RateLimitError:
                if attempt == self.max_retries - 1:
                    raise
                delay = self.rate_limit_delay * (2 ** attempt)
                time.sleep(delay)
```

## 缓存实现

### 1. 内存缓存 (src/utils/cache.py)

```python
import hashlib
import time
from typing import Dict, Any, Optional
from dataclasses import dataclass
from functools import wraps

@dataclass
class CacheEntry:
    """缓存条目"""
    data: Any
    timestamp: float
    ttl: int

class MemoryCache:
    """内存缓存实现"""

    def __init__(self, default_ttl: int = 3600):
        self.cache: Dict[str, CacheEntry] = {}
        self.default_ttl = default_ttl

    def _get_key(self, key: str) -> str:
        """生成缓存键"""
        return hashlib.md5(key.encode()).hexdigest()

    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        cache_key = self._get_key(key)
        entry = self.cache.get(cache_key)

        if entry and time.time() - entry.timestamp < entry.ttl:
            return entry.data
        elif entry:
            del self.cache[cache_key]

        return None

    def set(self, key: str, value: Any, ttl: int = None) -> None:
        """设置缓存"""
        cache_key = self._get_key(key)
        ttl = ttl or self.default_ttl

        self.cache[cache_key] = CacheEntry(
            data=value,
            timestamp=time.time(),
            ttl=ttl
        )

    def delete(self, key: str) -> None:
        """删除缓存"""
        cache_key = self._get_key(key)
        if cache_key in self.cache:
            del self.cache[cache_key]

    def clear(self) -> None:
        """清空缓存"""
        self.cache.clear()

def cache_result(ttl: int = 3600):
    """缓存装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 生成缓存键
            key_str = f"{func.__name__}_{str(args)}_{str(kwargs)}"

            cache = MemoryCache()
            result = cache.get(key_str)

            if result is not None:
                return result

            result = func(*args, **kwargs)
            cache.set(key_str, result, ttl)
            return result

        return wrapper
    return decorator
```

```python
import redis
from typing import Optional
import json
import pickle

class RedisCache:
    """Redis 缓存实现"""

    def __init__(self, host: str = "localhost", port: int = 6379, db: int = 0):
        self.redis = redis.Redis(host=host, port=port, db=db, decode_responses=True)

    def get(self, key: str) -> Optional[Any]:
        """获取缓存"""
        try:
            data = self.redis.get(key)
            if data:
                return json.loads(data)
            return None
        except Exception:
            return None

    def set(self, key: str, value: Any, ttl: int = 3600) -> None:
        """设置缓存"""
        try:
            self.redis.setex(key, ttl, json.dumps(value))
        except Exception:
            pass

    def delete(self, key: str) -> None:
        """删除缓存"""
        self.redis.delete(key)
```

```python
import logging
import logging.handlers
import os
from datetime import datetime

def setup_logger(name: str, log_file: str = None) -> logging.Logger:
    """设置日志"""
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)

    # 控制台处理器
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.INFO)

    # 文件处理器
    if log_file:
        file_handler = logging.handlers.RotatingFileHandler(
            log_file,
            maxBytes=10*1024*1024,  # 10MB
            backupCount=5
        )
        file_handler.setLevel(logging.INFO)

        # 文件格式
        file_formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
        )
        file_handler.setFormatter(file_formatter)

        logger.addHandler(file_handler)

    # 控制台格式
    console_formatter = logging.Formatter(
        '%(levelname)s - %(message)s'
    )
    console_handler.setFormatter(console_formatter)
    logger.addHandler(console_handler)

    return logger

class APILogger:
    """API 日志记录器"""

    def __init__(self):
        self.logger = setup_logger("claude_api")

    def log_request(self, model: str, prompt: str, tokens: int):
        """记录请求"""
        self.logger.info(f"Request - Model: {model}, Tokens: {tokens}, Prompt length: {len(prompt)}")

    def log_response(self, model: str, response: str, tokens: int, latency: float):
        """记录响应"""
        self.logger.info(f"Response - Model: {model}, Tokens: {tokens}, Latency: {latency:.2f}s")

    def log_error(self, error: Exception, context: str):
        """记录错误"""
        self.logger.error(f"Error in {context}: {str(error)}")
```

### 2. 性能监控

```python
import time
from functools import wraps
from typing import Callable

def monitor_performance(func: Callable) -> Callable:
    """性能监控装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start_time = time.time()

        try:
            result = func(*args, **kwargs)
            latency = time.time() - start_time

            # 记录性能指标
            if hasattr(func, '__self__'):
                logger = func.__self__.logger
            else:
                from .logger import APILogger
                logger = APILogger()

            logger.logger.info(f"{func.__name__} completed in {latency:.2f}s")

            return result
        except Exception as e:
            latency = time.time() - start_time
            logger = APILogger()
            logger.log_error(e, f"{func.__name__} (failed after {latency:.2f}s)")
            raise

    return wrapper
```

## 服务层实现

### 1. 聊天服务 (src/services/chat_service.py)

```python
from typing import List, Dict, Any
from ..api.client import ClaudeClient
from ..utils.cache import cache_result
from ..utils.logger import APILogger
from ..models.message import Message

class ChatService:
    """聊天服务"""

    def __init__(self, api_key: str):
        self.client = ClaudeClient(api_key)
        self.logger = APILogger()

    @cache_result(ttl=1800)  # 30分钟缓存
    def chat(
        self,
        messages: List[Dict[str, str]],
        system: str = None,
        max_tokens: int = 1000,
        temperature: float = 0.7
    ) -> Dict[str, Any]:
        """聊天功能"""
        try:
            # 构建提示词
            conversation = "\n".join([f"{msg['role']}: {msg['content']}" for msg in messages])

            # 记录请求
            self.logger.log_request(
                model=self.client.model,
                prompt=conversation,
                tokens=max_tokens
            )

            # 调用 API
            start_time = time.time()
            response = self.client.create_message(
                content=conversation,
                system=system,
                max_tokens=max_tokens,
                temperature=temperature
            )

            # 记录响应
            latency = time.time() - start_time
            self.logger.log_response(
                model=self.client.model,
                response=response['content'],
                tokens=response['usage']['output_tokens'],
                latency=latency
            )

            return response

        except Exception as e:
            self.logger.log_error(e, "chat")
            raise

    def chat_stream(
        self,
        messages: List[Dict[str, str]],
        system: str = None,
        max_tokens: int = 1000,
        temperature: float = 0.7
    ) -> str:
        """流式聊天"""
        try:
            conversation = "\n".join([f"{msg['role']}: {msg['content']}" for msg in messages])

            response = self.client.create_message(
                content=conversation,
                system=system,
                max_tokens=max_tokens,
                temperature=temperature,
                stream=True
            )

            return response['content']

        except Exception as e:
            self.logger.log_error(e, "chat_stream")
            raise
```

## 安全最佳实践

### 1. API 密钥管理

```python
import os
from cryptography.fernet import Fernet

class SecureConfig:
    """安全配置管理"""

    def __init__(self, key_file: str = ".key"):
        self.key_file = key_file
        self.cipher = self._load_key()

    def _load_key(self):
        """加载加密密钥"""
        if os.path.exists(self.key_file):
            with open(self.key_file, 'rb') as f:
                key = f.read()
        else:
            key = Fernet.generate_key()
            with open(self.key_file, 'wb') as f:
                f.write(key)
        return Fernet(key)

    def encrypt(self, data: str) -> str:
        """加密数据"""
        return self.cipher.encrypt(data.encode()).decode()

    def decrypt(self, encrypted_data: str) -> str:
        """解密数据"""
        return self.cipher.decrypt(encrypted_data.encode()).decode()
```

### 2. 输入验证

```python
from pydantic import BaseModel, validator, HttpUrl
from typing import List, Optional

class ChatRequest(BaseModel):
    """聊天请求模型"""
    messages: List[Dict[str, str]]
    max_tokens: int = 1000
    temperature: float = 0.7

    @validator('messages')
    def validate_messages(cls, v):
        """验证消息格式"""
        if not isinstance(v, list):
            raise ValueError('Messages must be a list')

        for msg in v:
            if not isinstance(msg, dict):
                raise ValueError('Each message must be a dictionary')
            if 'role' not in msg or 'content' not in msg:
                raise ValueError('Each message must have role and content')
            if msg['role'] not in ['user', 'assistant', 'system']:
                raise ValueError('Invalid role')

        return v

    @validator('max_tokens')
    def validate_max_tokens(cls, v):
        """验证令牌数量"""
        if v < 1 or v > 10000:
            raise ValueError('max_tokens must be between 1 and 10000')
        return v

    @validator('temperature')
    def validate_temperature(cls, v):
        """验证温度参数"""
        if v < 0 or v > 1:
            raise ValueError('temperature must be between 0 and 1')
        return v
```
