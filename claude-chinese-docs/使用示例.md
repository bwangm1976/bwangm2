# Claude API 使用示例

本章节提供了 Claude API 的各种使用示例，涵盖常见应用场景。

## 基础示例

### 1. 简单对话

```python
import anthropic

client = anthropic.Anthropic(api_key="your-api-key")

# 基本对话
response = client.messages.create(
    model="claude-3-sonnet-20240229",
    max_tokens=1000,
    messages=[
        {"role": "user", "content": "你好，请介绍一下 Claude。"}
    ]
)

print(response.content[0].text)
```


## 高级示例

### 1. 使用工具的聊天机器人

```python
import json
from anthropic import Anthropic

# 创建计算器工具
calculator_tool = {
    "name": "calculator",
    "description": "A simple calculator for mathematical operations",
    "input_schema": {
        "type": "object",
        "properties": {
            "expression": {
                "type": "string",
                "description": "数学表达式，如 '2 + 2' 或 'sqrt(16)'"
            }
        },
        "required": ["expression"]
    }
}

class SmartAssistant:
    def __init__(self, api_key):
        self.client = Anthropic(api_key=api_key)
        self.tools = [calculator_tool]

    def process_query(self, user_input):
        response = self.client.messages.create(
            model="claude-3-sonnet-20240229",
            max_tokens=1000,
            tools=self.tools,
            messages=[{"role": "user", "content": user_input}]
        )

        # 处理工具调用
        if response.tool_calls:
            for tool_call in response.tool_calls:
                if tool_call.function.name == "calculator":
                    result = self.calculate(tool_call.function.arguments)
                    return f"计算结果：{result}"

        return response.content[0].text

    def calculate(self, expression):
        """简单的计算器函数"""
        try:
            # 安全评估数学表达式
            allowed_chars = set('0123456789+-*/.() ')
            if not all(c in allowed_chars for c in expression):
                return "不安全的数学表达式"

            result = eval(expression)
            return str(result)
        except:
            return "计算错误"

# 使用示例
assistant = SmartAssistant("your-api-key")
print(assistant.process_query("2 + 2 等于多少？"))
```

### 2. 文档问答系统

```python
class DocumentQA:
    def __init__(self, api_key):
        self.client = Anthropic(api_key=api_key)
        self.documents = []

    def add_document(self, title, content):
        """添加文档"""
        self.documents.append({
            "title": title,
            "content": content
        })

    def ask_question(self, question):
        """回答关于文档的问题"""
        # 构建提示词，包含所有文档内容
        context = ""
        for doc in self.documents:
            context += f"文档：{doc['title']}\n{doc['content']}\n\n"

        response = self.client.messages.create(
            model="claude-3-sonnet-20240229",
            max_tokens=1000,
            messages=[
                {
                    "role": "user",
                    "content": f"基于以下文档内容回答问题：\n\n{context}\n\n问题：{question}"
                }
            ]
        )

        return response.content[0].text

# 使用示例
qa = DocumentQA("your-api-key")
qa.add_document("Python 简介", "Python 是一种高级编程语言...")
qa.add_document("机器学习基础", "机器学习是人工智能的一个分支...")

answer = qa.ask_question("什么是机器学习？")
print(answer)
```

## 性能优化示例

### 1. 批量处理

```python
import concurrent.futures
from anthropic import Anthropic

def batch_process(prompts, api_key, max_workers=5):
    """批量处理多个提示"""
    client = Anthropic(api_key=api_key)
    results = []

    def process_single_prompt(prompt):
        response = client.messages.create(
            model="claude-3-sonnet-20240229",
            max_tokens=1000,
            messages=[{"role": "user", "content": prompt}]
        )
        return response.content[0].text

    with concurrent.futures.ThreadPoolExecutor(max_workers=max_workers) as executor:
        future_to_prompt = {executor.submit(process_single_prompt, prompt): prompt for prompt in prompts}
        for future in concurrent.futures.as_completed(future_to_prompt):
            result = future.result()
            results.append(result)

    return results

# 使用示例
prompts = ["什么是人工智能？", "机器学习的基本概念", "深度学习的应用"]
results = batch_process(prompts, "your-api-key")
```

### 2. 缓存机制

```python
import hashlib
import time
from functools import lru_cache
from anthropic import Anthropic

class CachedAPI:
    def __init__(self, api_key, cache_ttl=3600):
        self.client = Anthropic(api_key=api_key)
        self.cache = {}
        self.cache_ttl = cache_ttl

    def _get_cache_key(self, prompt, model):
        """生成缓存键"""
        key_string = f"{prompt}_{model}"
        return hashlib.md5(key_string.encode()).hexdigest()

    def _is_cache_valid(self, cache_entry):
        """检查缓存是否有效"""
        return time.time() - cache_entry['timestamp'] < self.cache_ttl

    def query(self, prompt, model="claude-3-sonnet-20240229"):
        """带缓存的查询"""
        cache_key = self._get_cache_key(prompt, model)

        # 检查缓存
        if cache_key in self.cache and self._is_cache_valid(self.cache[cache_key]):
            return self.cache[cache_key]['response']

        # 执行查询
        response = self.client.messages.create(
            model=model,
            max_tokens=1000,
            messages=[{"role": "user", "content": prompt}]
        )

        # 缓存结果
        self.cache[cache_key] = {
            'response': response.content[0].text,
            'timestamp': time.time()
        }

        return response.content[0].text

# 使用示例
cached_api = CachedAPI("your-api-key")
response = cached_api.query("什么是机器学习？")
```

## 错误处理示例

```python
from anthropic import APIError, RateLimitError, AuthenticationError
import time

def safe_api_call(func, max_retries=3, initial_delay=1):
    """带重试机制的API调用"""
    for attempt in range(max_retries):
        try:
            return func()
        except AuthenticationError:
            print("认证失败，请检查API密钥")
            raise
        except RateLimitError:
            if attempt < max_retries - 1:
                delay = initial_delay * (2 ** attempt)
                print(f"请求过于频繁，{delay}秒后重试...")
                time.sleep(delay)
                continue
            else:
                print("达到最大重试次数")
                raise
        except APIError as e:
            print(f"API错误: {e}")
            raise

# 使用示例
try:
    response = safe_api_call(
        lambda: client.messages.create(
            model="claude-3-sonnet-20240229",
            max_tokens=1000,
            messages=[{"role": "user", "content": "Hello"}]
        )
    )
except Exception as e:
    print(f"最终错误: {e}")
```
